1  如果两个点AB，在统一条线上则c点在AB点的同轴上加一或者减一
$$
$$ 如果两个点不同轴，则选一点的x轴，选一点的y轴 $$
$$$$
```cpp
struct BIT
{
    int tree[maxn];
    void cha(int x,int y)
    {
        for(int i=x;i<=n;i+=i&-i)
            tree[i]+=y;
    }
    void cha(int l,int r,int x)
    {
        cha(l,x);
        cha(r+1,-x);
    }
    int fi(int x)
    {
        int ret=0;
        for(int i=x;i;i-=i&-i)
            ret+=tree[i];
        return ret;
    }
    int fi(int l,int r)
    {
        return fi(r)-fi(l-1);
    }
}bit;
```
